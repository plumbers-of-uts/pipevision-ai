```dart
    final file = File(p.join(dir.path, 'app.sqlite'));
```

```dart
// lib/features/todos/data/local/todos_dao.dart
import 'package:drift/drift.dart';
import '../../../../core/database/app_database.dart';

part 'todos_dao.g.dart';

@DriftAccessor(tables: [TodosTable])
class TodosDao extends DatabaseAccessor<AppDatabase> with _$TodosDaoMixin {
  TodosDao(super.db);

  /// Live stream — emits updated rows whenever Drift detects a change.
  Stream<List<TodosTableData>> watchAll() => select(todosTable).watch();

  /// Upsert full list from network (replaces stale rows).
  Future<void> upsertAll(List<TodosTableCompanion> rows) =>
      batch((b) => b.insertAllOnConflictUpdate(todosTable, rows));

  /// Upsert a single row after create/toggle.
  Future<void> upsertOne(TodosTableCompanion row) =>
      into(todosTable).insertOnConflictUpdate(row);

  Future<void> deleteById(int id) =>
      (delete(todosTable)..where((t) => t.id.equals(id))).go();

  Future<TodosTableData?> findById(int id) =>
      (select(todosTable)..where((t) => t.id.equals(id))).getSingleOrNull();
}
```

### §3d. Remote Data Source (data layer — Dio calls)

```dart
// lib/features/todos/data/remote/todos_remote_data_source.dart
import 'package:dio/dio.dart';
import '../dtos/todo_dto.dart';

/// All Dio calls for the /todos resource.
/// No caching here — transport concerns only.
class TodosRemoteDataSource {
  TodosRemoteDataSource(this._dio);
  final Dio _dio;

  Future<List<TodoDto>> fetchAll() async {
    final response = await _dio.get<List<dynamic>>('/todos');
    return (response.data ?? [])
        .map((json) => TodoDto.fromJson(json as Map<String, dynamic>))
        .toList();
  }

  Future<TodoDto> create(String title) async {
    final response = await _dio.post<Map<String, dynamic>>(
      '/todos',
      data: {'title': title},
    );
    return TodoDto.fromJson(response.data!);
  }

  Future<TodoDto> toggle(int id) async {
    final response = await _dio.patch<Map<String, dynamic>>('/todos/$id/toggle');
    return TodoDto.fromJson(response.data!);
  }

  Future<void> delete(int id) => _dio.delete<void>('/todos/$id');
}
```

### §3e. Offline-First Repository Implementation (data layer)

```dart
// lib/features/todos/data/todo_repository_impl.dart
import 'package:drift/drift.dart';
import '../../../../core/database/app_database.dart';
import '../domain/todo.dart';
import '../domain/todo_repository.dart';
import 'dtos/todo_dto.dart';
import 'local/todos_dao.dart';
import 'remote/todos_remote_data_source.dart';

/// TTL for the todos list cache. Revalidate if the newest row is older than this.
const _kCacheTtl = Duration(minutes: 5);

/// Offline-first implementation of [TodoRepository].
///
/// Caching contract:
///   - [watchAll] maps the Drift stream to domain entities and emits immediately.
///     A background revalidation triggers whenever the cache is stale (> TTL)
///     or empty. Drift re-emits the stream automatically after each upsert.
///   - Writes ([create], [toggle], [delete]) hit the network first, then upsert
///     or delete the local Drift row so the stream emits the change instantly.
///   - Cache key = logical operation, never URLs. TTL is explicit; no infinite cache.
///   - Secrets and user-authored data are NOT stored here — use flutter_secure_storage
///     or a separate Drift table with a different retention policy.
class TodoRepositoryImpl implements TodoRepository {
  TodoRepositoryImpl({
    required TodosDao dao,
    required TodosRemoteDataSource remote,
  })  : _dao = dao,
        _remote = remote;

  final TodosDao _dao;
  final TodosRemoteDataSource _remote;

  // ---------------------------------------------------------------------------
  // READ — stale-while-revalidate
  // ---------------------------------------------------------------------------

  @override
  Stream<List<Todo>> watchAll() async* {
    // 1. Emit cached rows immediately (may be empty on first run).
    final initial = await _dao.watchAll().first;
    final now = DateTime.now();

    // 2. Determine whether revalidation is needed.
    final isStale = initial.isEmpty ||
        initial.any((row) => now.difference(row.cachedAt) > _kCacheTtl);

    if (isStale) {
      // 3. Fetch from network in the background and upsert into Drift.
      // Drift's stream will re-emit automatically after the upsert.
      _revalidate().ignore();
    }

    // 4. Forward the live Drift stream (emits on every upsert from step 3 onward).
    yield* _dao.watchAll().map(
          (rows) => rows.map(_rowToEntity).toList(),
        );
  }

  Future<void> _revalidate() async {
    try {
      final dtos = await _remote.fetchAll();
      final now = DateTime.now();
      await _dao.upsertAll(
        dtos.map((dto) => _dtoToCompanion(dto, cachedAt: now)).toList(),
      );
    } catch (_) {
      // Network failure is non-fatal when the cache is warm.
      // The existing stream emission (stale data) stands.
    }
  }

  // ---------------------------------------------------------------------------
  // WRITE — invalidate/update cache after success
  // ---------------------------------------------------------------------------

  @override
  Future<Todo> create(String title) async {
    final dto = await _remote.create(title);
    final now = DateTime.now();
    // Upsert the new row so watchAll() emits without waiting for a revalidation.
    await _dao.upsertOne(_dtoToCompanion(dto, cachedAt: now));
    return dto.toDomain(cachedAt: now);
  }

  @override
  Future<Todo> toggle(int id) async {
    final dto = await _remote.toggle(id);
    final now = DateTime.now();
    await _dao.upsertOne(_dtoToCompanion(dto, cachedAt: now));
    return dto.toDomain(cachedAt: now);
  }

  @override
  Future<void> delete(int id) async {
    await _remote.delete(id);
    // Remove the row so the stream emits the deletion immediately.
    await _dao.deleteById(id);
  }

  // ---------------------------------------------------------------------------
  // Helpers
  // ---------------------------------------------------------------------------

  Todo _rowToEntity(TodosTableData row) => Todo(
        id: row.id,
        title: row.title,
        completed: row.completed,
        cachedAt: row.cachedAt,
      );

  TodosTableCompanion _dtoToCompanion(TodoDto dto, {required DateTime cachedAt}) =>
      TodosTableCompanion(
        id: Value(dto.id),
        title: Value(dto.title),
        completed: Value(dto.completed),
        cachedAt: Value(cachedAt),
      );
}
```

## §5. Dio Client with Auth + Retry + Logging Interceptors

```dart
// lib/core/network/dio_client.dart
import 'package:dio/dio.dart';
import 'package:flutter/foundation.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:flutter_secure_storage/flutter_secure_storage.dart';
import 'package:pretty_dio_logger/pretty_dio_logger.dart';

/// Builds the app-wide Dio instance with interceptors.
/// Called once from the DI layer; never instantiated in features directly.
Dio buildDioClient(Ref ref) {
  final dio = Dio(
    BaseOptions(
      baseUrl: const String.fromEnvironment(
        'API_BASE_URL',
        defaultValue: 'https://api.example.com',
      ),
      connectTimeout: const Duration(seconds: 15),
      receiveTimeout: const Duration(seconds: 30),
      headers: {'Content-Type': 'application/json'},
    ),
  );

  // Transport-layer interceptors only — no caching here.
  dio.interceptors.addAll([
    AuthInterceptor(const FlutterSecureStorage()),
    RetryInterceptor(dio),
    if (kDebugMode) PrettyDioLogger(requestBody: true, responseBody: true),
  ]);

  return dio;
}

// ---------------------------------------------------------------------------
// Auth interceptor — injects Bearer token from secure storage
// ---------------------------------------------------------------------------

class AuthInterceptor extends Interceptor {
  AuthInterceptor(this._storage);
  final FlutterSecureStorage _storage;

  @override
  Future<void> onRequest(
    RequestOptions options,
    RequestInterceptorHandler handler,
  ) async {
    final token = await _storage.read(key: 'access_token');
    if (token != null) {
      options.headers['Authorization'] = 'Bearer $token';
    }
    handler.next(options);
  }

  @override
  void onError(DioException err, ErrorInterceptorHandler handler) {
    if (err.response?.statusCode == 401) {
      // TODO: trigger token refresh or sign-out flow via ref
      // For now, pass the error through to the caller.
    }
    handler.next(err);
  }
}

// ---------------------------------------------------------------------------
// Retry interceptor — exponential back-off on transient failures
// ---------------------------------------------------------------------------

class RetryInterceptor extends Interceptor {
  RetryInterceptor(this._dio, {this.maxRetries = 3});
  final Dio _dio;
  final int maxRetries;

  @override
  Future<void> onError(
    DioException err,
    ErrorInterceptorHandler handler,
  ) async {
    final attempt = err.requestOptions.extra['_retryCount'] as int? ?? 0;
    final isRetriable = err.type == DioExceptionType.connectionTimeout ||
        err.type == DioExceptionType.receiveTimeout ||
        err.type == DioExceptionType.connectionError;

    if (isRetriable && attempt < maxRetries) {
      await Future.delayed(Duration(milliseconds: 200 * (1 << attempt)));
      try {
        final opts = err.requestOptions
          ..extra['_retryCount'] = attempt + 1;
        final response = await _dio.fetch<dynamic>(opts);
        handler.resolve(response);
        return;
      } catch (_) {
        // Fall through to next handler on retry failure.
      }
    }
    handler.next(err);
  }
}
```

## §6. Todos Screen (ConsumerStatefulWidget)

```dart
// lib/features/todos/presentation/screens/todos_screen.dart
import 'package:flutter/material.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';
import '../../domain/todo.dart';
import '../providers/todos_provider.dart';
import '../widgets/todo_list_item.dart';
import '../../../../shared/widgets/empty_state.dart';
import '../../../../shared/widgets/error_view.dart';

class TodosScreen extends ConsumerStatefulWidget {
  const TodosScreen({super.key});

  @override
  ConsumerState<TodosScreen> createState() => _TodosScreenState();
}

class _TodosScreenState extends ConsumerState<TodosScreen> {
  final _titleController = TextEditingController();

  @override
  void dispose() {
    _titleController.dispose();   // always dispose controllers
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    final todosAsync = ref.watch(todosNotifierProvider);

    return Scaffold(
      appBar: AppBar(
        title: const Text('Todos'),
        actions: [
          IconButton(
            icon: const Icon(Icons.refresh),
            tooltip: 'Refresh',
            // Invalidate causes build() to re-subscribe to watchAll()
            // which triggers a fresh revalidation cycle.
            onPressed: () => ref.invalidate(todosNotifierProvider),
          ),
        ],
      ),
      body: todosAsync.when(
        // Warm cache is available — render immediately, no spinner.
        // Riverpod passes skipLoadingOnRefresh: true by default in Riverpod 2.x
        // so a background revalidation does not re-show the loading state.
        data: (todos) => todos.isEmpty
            ? const EmptyState(
                icon: Icons.check_circle_outline,
                title: 'No todos yet',
                subtitle: 'Add your first todo below.',
              )
            : _TodoList(todos: todos),
        loading: () => const Center(child: CircularProgressIndicator()),
        error: (err, _) => ErrorView(
          message: err.toString(),
          onRetry: () => ref.invalidate(todosNotifierProvider),
        ),
      ),
      floatingActionButton: FloatingActionButton(
        onPressed: _showAddDialog,
        tooltip: 'Add todo',
        child: const Icon(Icons.add),
      ),
    );
  }

  void _showAddDialog() {
    _titleController.clear();
    showDialog<void>(
      context: context,
      builder: (ctx) => AlertDialog(
        title: const Text('New Todo'),
        content: TextField(
          controller: _titleController,
          autofocus: true,
          decoration: const InputDecoration(
            labelText: 'Title',
            hintText: 'What needs to be done?',
          ),
          onSubmitted: (_) => _submitAdd(ctx),
        ),
        actions: [
          TextButton(
            onPressed: () => Navigator.of(ctx).pop(),
            child: const Text('Cancel'),
          ),
          FilledButton(
            onPressed: () => _submitAdd(ctx),
            child: const Text('Add'),
          ),
        ],
      ),
    );
  }

  Future<void> _submitAdd(BuildContext dialogCtx) async {
    final title = _titleController.text.trim();
    if (title.isEmpty) return;
    Navigator.of(dialogCtx).pop();
    await ref.read(todosNotifierProvider.notifier).create(title);
  }
}

// ---------------------------------------------------------------------------
// Internal list widget
// ---------------------------------------------------------------------------

class _TodoList extends ConsumerWidget {
  const _TodoList({required this.todos});
  final List<Todo> todos;

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    return RefreshIndicator(
      // Pull-to-refresh forces a full revalidation cycle.
      onRefresh: () async => ref.invalidate(todosNotifierProvider),
      child: ListView.builder(
        itemCount: todos.length,
        itemBuilder: (ctx, i) => TodoListItem(
          todo: todos[i],
          onToggle: () =>
              ref.read(todosNotifierProvider.notifier).toggle(todos[i].id),
          onDelete: () =>
              ref.read(todosNotifierProvider.notifier).delete(todos[i].id),
        ),
      ),
    );
  }
}
```

## §10. Cache Rules Recap

> These rules are mandatory. Violating any of them is a defect, not a style choice.

1. **Repository layer, not transport.** Cache decoded `Todo` domain entities inside `TodoRepositoryImpl`. Do NOT use a Dio interceptor (e.g. `dio_http_cache`, `dio_cache_interceptor`) as the system of record for domain reads — interceptors operate on raw bytes and are invisible to the domain and its tests.
2. **Stale-while-revalidate.** `watchAll()` emits the locally-cached Drift rows immediately (zero latency for the user), then silently revalidates from the network and re-emits via the Drift stream. No loading spinner on cache hits.
3. **Explicit TTL.** Every cache row stores `cachedAt`. The Repository compares it against `_kCacheTtl` to decide whether to revalidate. Never use an implicit or infinite TTL.
4. **Key = operation + params, never URLs.** The Drift table primary key (`id`) and the DAO operation name (e.g., `watchAll`, `findById`) are the effective cache keys. Changing the backend URL does not invalidate the cache; only writing new data or deleting rows does.
5. **Invalidate on write.** `create` → upsert new row. `toggle` → upsert updated row. `delete` → delete row. After each mutation the Drift stream emits the new state to all active subscribers without any explicit refresh call in the UI.
6. **Boundaries.** The Drift cache table is for transient server-owned data. Durable user-authored data (drafts, settings) → separate Drift table with its own retention policy. Secrets (tokens) → `flutter_secure_storage`. Do not use the cache table as a system of record.
